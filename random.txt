   @Transactional
    public void saveValidatedJson(String uid, List<GppFieldValidationResponse> responseList) {
        String query = "UPDATE rx_project_sot SET validated_json = :validatedJson WHERE sotgppid_mapping = :uid";
        try {
            String jsonResponse = objectMapper.writeValueAsString(responseList);
            entityManager.createNativeQuery(query)
                    .setParameter("validatedJson", jsonResponse)
                    .setParameter("uid", uid)
                    .executeUpdate();
        } catch (JsonProcessingException e) {
            logger.error("Error converting responseList to JSON: {}", e.getMessage());
            throw new RuntimeException("Error converting responseList to JSON", e);
        } catch (Exception e) {
            logger.error("Error updating validated_json for uid {}: {}", uid, e.getMessage());
            logger.debug("Exception stack trace", e);
            throw new RuntimeException("Error updating validated_json", e);
        }
    }
