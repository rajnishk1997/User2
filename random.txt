@Service
public class UserService {

    @Autowired
    private UserDao userDao;

    public List<Integer> getUserIdsByManagerId(Integer managerId, boolean isHierarchy) {
        List<Integer> userIds = new ArrayList<>();

        // Fetch the manager itself
        User manager = userDao.findById(managerId).orElse(null);
        if (manager != null && hasAuditorRole(manager)) {
            userIds.add(manager.getUserRid()); // Add manager's userId if it has 'Auditor' role
        }

        List<User> users;
        if (isHierarchy) {
            users = fetchUsersInHierarchy(managerId);
        } else {
            users = userDao.findImmediateAuditorsUnderManager(managerId);
        }

        // Add the userIds of the fetched users to the list
        userIds.addAll(users.stream().map(User::getUserRid).collect(Collectors.toList()));

        return userIds;
    }

    private List<User> fetchUsersInHierarchy(Integer managerId) {
        List<User> allAuditors = new ArrayList<>();
        Queue<Integer> queue = new LinkedList<>();
        queue.add(managerId);

        while (!queue.isEmpty()) {
            Integer currentManagerId = queue.poll();
            List<User> immediateUsers = userDao.findAuditorsUnderManagerHierarchy(currentManagerId);

            for (User user : immediateUsers) {
                if (!allAuditors.contains(user)) {
                    allAuditors.add(user);
                    if (user.getUserRoles().stream().anyMatch(role -> "Manager".equals(role.getRoleName()))) {
                        queue.add(user.getUserRid());
                    }
                }
            }
        }
        return allAuditors;
    }

    private boolean hasAuditorRole(User user) {
        return user.getUserRoles().stream().anyMatch(role -> "Auditor".equals(role.getRoleName()));
    }
}





@Query("SELECT u.userRid FROM User u WHERE u.isNewUser = false AND " +
       "(u.manager.userRid = :managerId AND EXISTS (SELECT r FROM u.userRoles ur JOIN ur.role r WHERE r.roleName = 'Auditor')) " +
       "OR (u.userRid = :managerId AND EXISTS (SELECT r FROM u.userRoles ur JOIN ur.role r WHERE r.roleName = 'Auditor'))")
List<Integer> findAuditorUserIdsUnderManagerHierarchy(@Param("managerId") Integer managerId);



@Query("SELECT u.userRid FROM User u WHERE u.isNewUser = false AND " +
       "(u.manager.userRid = :managerId AND EXISTS (SELECT r FROM u.userRoles ur JOIN ur.role r WHERE r.roleName = 'Auditor')) " +
       "OR (u.userRid = :managerId AND EXISTS (SELECT r FROM u.userRoles ur JOIN ur.role r WHERE r.roleName = 'Auditor'))")
List<Integer> findImmediateAuditorUserIdsUnderManager(@Param("managerId") Integer managerId);
