public RoleInfo getPermissionsForRole(int roleId) {
        Role role = roleRepository.findById(roleId).orElseThrow(() -> new RuntimeException("Role not found"));
        Set<PermissionInfo> permissionInfos = role.getRolePermissions().stream()
                .map(rolePermission -> new PermissionInfo(rolePermission.getPermission().getPermissionRid(), rolePermission.getPermission().getPermissionName()))
                .collect(Collectors.toSet());

        return new RoleInfo(role.getRoleRid(), role.getRoleName(), permissionInfos);
    }
	
	
	 @GetMapping("/{roleId}/permissions")
    public ResponseEntity<RoleInfo> getPermissionsForRole(@PathVariable int roleId) {
        RoleInfo roleInfo = rolePermissionService.getPermissionsForRole(roleId);
        return new ResponseEntity<>(roleInfo, HttpStatus.OK);
    }
