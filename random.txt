public interface ProjectEventDetailsRepository extends JpaRepository<ProjectEventDetails, Long> {

    // Fetch records by sotId and order by dateTime ascending
    @Query("SELECT p FROM ProjectEventDetails p WHERE p.sotId = :sotId ORDER BY p.dateTime ASC")
    List<ProjectEventDetails> findBySotIdOrderByDateTimeAsc(@Param("sotId") Long sotId);

    // Fetch records by createdById and order by sotId ascending
    @Query("SELECT p FROM ProjectEventDetails p WHERE p.createdbyId = :createdById ORDER BY p.sotId ASC")
    List<ProjectEventDetails> findByCreatedByIdOrderBySotIdAsc(@Param("createdById") Integer createdById);

    // Fetch all records ordered by sotId ascending
    @Query("SELECT p FROM ProjectEventDetails p ORDER BY p.sotId ASC")
    List<ProjectEventDetails> findAllByOrderBySotIdAsc();
}
