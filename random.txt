@PostMapping("/{roleId}/permissions")
public ResponseEntity<RoleInfo> mapPermissionsToRole(
        @PathVariable int roleId,
        @RequestBody Map<String, Object> requestBody) {

    try {
        int currentUserRid = (int) requestBody.get("currentUserRid"); // Extract currentUserRid from top level

        ObjectMapper mapper = new ObjectMapper();
        List<RoleInfo> roleInfoList = mapper.convertValue(requestBody.get("roles"), new TypeReference<List<RoleInfo>>() {});

        RoleInfo updatedRoleInfo = rolePermissionService.mapPermissionsToRole(roleId, roleInfoList, currentUserRid);

        return ResponseEntity.ok(updatedRoleInfo);
    } catch (Exception e) {
        // Handle exceptions
        return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR)
                             .body(null); // You can return an error response or handle differently as needed
    }
}
