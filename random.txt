User managerUser = userDao.findByUserName(managerMISName)
        .orElseGet(() -> {
            // Insert a new manager user with isActive = false if the manager is not found
            User newManagerUser = new User();
            newManagerUser.setUserName(managerMISName);
            newManagerUser.setActiveUser(false);
            return userDao.save(newManagerUser);
        });

// Set manager for the authenticated user
user.setManager(managerUser);
userDao.save(user);
