 try {
            List<ProjectEventDetails> projectEventDetails = service.getByProjectNo(projectNo);
            
            if (projectEventDetails.isEmpty()) {
                ReqRes reqRes = new ReqRes(404, "Not Found", "No events found for project number: " + projectNo);
                ResponseWrapper<List<ProjectEventDetails>> responseWrapper = new ResponseWrapper<>(null, reqRes);
                return new ResponseEntity<>(responseWrapper, HttpStatus.NOT_FOUND);
            }
            
            ReqRes reqRes = new ReqRes(200, null, "Success");
            ResponseWrapper<List<ProjectEventDetails>> responseWrapper = new ResponseWrapper<>(projectEventDetails, reqRes);
            return new ResponseEntity<>(responseWrapper, HttpStatus.OK);
        } catch (Exception ex) {
            ReqRes reqRes = new ReqRes(500, ex.getMessage(), "An unexpected error occurred.");
            ResponseWrapper<List<ProjectEventDetails>> responseWrapper = new ResponseWrapper<>(null, reqRes);
            return new ResponseEntity<>(responseWrapper, HttpStatus.INTERNAL_SERVER_ERROR);
        }
    }
